% Place un PDF à la position (x%, y%) sur la page
\newcommand{\placePDF}[5]{%
\begin{tikzpicture}[remember picture, overlay]
\node[anchor=#5] at ($(current page.south west) + (#1/100*\paperwidth, #2/100*\paperheight)$) {
\includegraphics[scale=#4]{#3}
        };
\end{tikzpicture}%
}

% Place un PDF dans une boîte avec titre
% #1 = x (%) depuis le bord gauche
% #2 = y (%) depuis le bord bas
% #3 = largeur de la boîte (% de \paperwidth)
% #4 = taille du texte du titre (scalebox)
% #5 = titre de la boîte
% #6 = fichier PDF
% #7 = ancre
\newcommand{\placePDFbox}[7]{%
\pgfmathsetmacro{\boxwidth}{#3/100*\paperwidth}%
\begin{tikzpicture}[remember picture, overlay]
  \node[
    anchor=#7,
    draw=black,
    line width=3pt,
    rounded corners=15pt,
    fill=white,
    inner sep=15pt,
    align=center,
    text width=\boxwidth pt,
  ] at ($(current page.south west) + (#1/100*\paperwidth, #2/100*\paperheight)$)
  {
    \scalebox{#4}{\textbf{#5}}\\[0.4cm]
    \includegraphics[width=\boxwidth pt]{#6}
  };
\end{tikzpicture}%
}

% Bulle avec titre, auteurs et organisation
% #1 = x (%) depuis le bord gauche
% #2 = y (%) depuis le bord bas
% #3 = scale
% #4 = taille de base du texte — titre = #4*1.7, auteurs = #4, organisation = #4*0.85
% #5 = Titre du poster
% #6 = Auteur 1
% #7 = Auteur 2
% #8 = Auteur 3
% #9 = Organisation
\newcommand{\posterBubble}[9]{%
\pgfmathsetmacro{\sizetitle}{#4 * 1.7}%
\pgfmathsetmacro{\sizeorg}{#4 * 0.85}%
\begin{tikzpicture}[remember picture, overlay]
  \node[
    anchor=center,
    draw=black,
    line width=4pt,
    rounded corners=30pt,
    fill=white,
    text=black,
    inner sep=25pt,
    align=center,
    scale=#3,
  ] at ($(current page.south west) + (#1/100*\paperwidth, #2/100*\paperheight)$)
  {
    \scalebox{\sizetitle}{\textbf{#5}}\\[1cm]
    \scalebox{#4}{#6 \quad\textbullet\quad #7 \quad\textbullet\quad #8}\\[1cm]
    \scalebox{\sizeorg}{\textit{#9}}
  };
\end{tikzpicture}%
}
